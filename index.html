import { encrypt, sign } from '@shaktisetu/crypto';

export const broadcastSOS = async (location: Coordinates) => {
  // Encrypt payload with Guardian's public keys
  const payload = encrypt(location, guardianKeys);
  const signature = sign(payload, userPrivateKey);

  await network.dispatch({ payload, signature });
};
